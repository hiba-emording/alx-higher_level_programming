# FILE: tests/1-my_list.txt

# MODULE: 1-my_list.py.
# DESCRIPTION: This module contains the MyList class, a subclass of list,
#              with a method to print a sorted list.

# CLASS: MyList
# DESCRIPTION:
#   - Contains a public method print_sorted to print the list in sorted order.

# USAGE:
#   - >>> MyList = __import__('1-my_list').MyList
#   - >>> my_list = MyList()
#   - >>> my_list.append()
#   - >>> my_list.print_sorted()
#     Output

# Test Cases:

>>> MyList = __import__('1-my_list').MyList

# Case 0: Empty List:
>>> my_list = MyList()
>>> print(my_list)
[]
>>> my_list.print_sorted()
[]

# Case 1: Print already sorted list:
>>> my_list = MyList()
>>> my_list.append(1)
>>> my_list.append(2)
>>> my_list.append(3)
>>> print(my_list)
[1, 2, 3]
>>> my_list.print_sorted()
[1, 2, 3]

Case 2: Print non sorted list:
>>> my_list = MyList()
>>> my_list.append(2)
>>> my_list.append(-3)
>>> my_list.append(1)
>>> my_list.append(-4)
>>> my_list.append(3)
>>> print(my_list)
[2, -3, 1, -4, 3]
>>> my_list.print_sorted()
[-4, -3, 1, 2, 3]

Case 3: A list with identical elements:
>>> my_list = MyList()
>>> my_list.append(-5)
>>> my_list.append(-5)
>>> my_list.append(-5)
>>> print(my_list)
[-5, -5, -5]
>>> my_list.print_sorted()
[-5, -5, -5]

Case 4: Wrong number of arguments:
>>> my_list.print_sorted(0)
Traceback (most recent call last):
TypeError: print_sorted() takes 1 positional argument but 2 were given
